pipeline {
    agent any

    stages {
        stage('Clone Code') {
            steps {
                git branch: 'main', url: 'https://github.com/crazyloke1-hash/devops-app.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                bat 'docker build -t lokesh1926/expense-tracker:v1 .'
            }
        }

        stage('Push to DockerHub') {
            steps {
                bat 'docker login -u lokesh1926 -p YOUR_PASSWORD'
                bat 'docker push lokesh1926/expense-tracker:v1'
            }
        }
    }
}
